if($_POST['mode'] == 'add'){
header("Location:index.php");
}
/*
然後這裡在進行縮圖，他就會在背景中執行
*/
if($_POST['mode'] == 'add'){
for($i = 1; $i < 6; $i++){
convert_sh($upload_floder,$file_name[$i],$small_size[$i]);
}
}
?>
